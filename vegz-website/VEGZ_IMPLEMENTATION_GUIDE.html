Fix Farmer Submit "Not Found"

Root Cause

The POST /api/v1/farmer/listings route did not exist in the backend. The 404 "Not Found" error was correct — the endpoint was never created.

Files to add (in the zip)

  • backend/src/routes/farmer.routes.js — new farmer API
  • backend/src/routes/admin.routes.js — new admin API
  • database/003_farmer_admin_tables.sql — new tables
  • frontend/pages/admin.html — admin dashboard

Step-by-step fix

1

Run new SQL on Server 2 (MySQL)

SSH to Server 2 and run:

mysql -u vegz_user -p vegz_db < database/003_farmer_admin_tables.sql
2

Upload new backend files to Server 1

scp -i key.pem backend/src/routes/farmer.routes.js ubuntu@SERVER1_IP:/var/www/vegz/backend/src/routes/
scp -i key.pem backend/src/routes/admin.routes.js ubuntu@SERVER1_IP:/var/www/vegz/backend/src/routes/

Restart backend

pm2 restart vegz-api && pm2 logs vegz-api --lines 20
4

Test the endpoint

# Should return: {"success":false,"message":"Admin not authenticated"}

Push Notifications — FREE with Firebase FCM

✅ Firebase Cloud Messaging (FCM) is completely FREE for unlimited notifications. No credit card. No cost ever.

What you get

  • Customer gets push notification when order status changes (confirmed, out for delivery, delivered)
  • Farmer gets notification when agent is assigned or payment is sent
  • Admin gets notification when new order is placed or farmer submits listing
  • Works on Chrome, Firefox, Android (web push) and native Android app (if you build it later)

Step 2 — Add to Frontend

Create frontend/js/push.js:

// VEGZ Push Notifications via Firebase FCM
const firebaseConfig = {
  apiKey: "AIzaSy...",           // from Firebase console
  authDomain: "vegz.firebaseapp.com",
  projectId: "vegz-notifications",
  messagingSenderId: "123456789012",
  appId: "1:123456789012:web:abcdef"
};

// Initialize Firebase
import { initializeApp } from "https://www.gstatic.com/firebasejs/10.7.0/firebase-app.js";
import { getMessaging, getToken, onMessage } from "https://www.gstatic.com/firebasejs/10.7.0/firebase-messaging.js";

const app       = initializeApp(firebaseConfig);
const messaging = getMessaging(app);

// Ask user for notification permission
async function requestPushPermission() {
  try {
    const permission = await Notification.requestPermission();
    if (permission !== 'granted') return;

    const token = await getToken(messaging, {
      vapidKey: 'YOUR_VAPID_KEY' // from Firebase Console → Cloud Messaging → Web config
    });

    // Save token to backend
    await fetch(`${window.VEGZ_CONFIG.API_BASE}/api/v1/users/push-token`, {
      method: 'POST',
      headers: { 'Content-Type':'application/json', 'Authorization':'Bearer '+localStorage.getItem('vegz_token') },
      body: JSON.stringify({ token, platform: 'web' })
    });

    console.log('Push notifications enabled!');
  } catch (err) {
    console.log('Push permission denied or error:', err);
  }
}

// Handle foreground notifications
onMessage(messaging, (payload) => {
  const { title, body } = payload.notification;
  // Show as in-app toast
  showToast(`${title}: ${body}`, 'info');
  // Also show browser notification
  new Notification(title, { body, icon: '/icon.png' });
});

// Call this after login
export { requestPushPermission };

Step 3 — Send Notifications from Backend

In your order controller, after status update, add:

// backend/src/services/notification.service.js
const axios = require('axios');

async function sendPushToUser(userId, title, body, data = {}) {
  const { pool } = require('../config/database');
  const [tokens] = await pool.execute(
    'SELECT token FROM push_tokens WHERE user_id=? AND platform IN ("web","android")', [userId]
  );
  if (!tokens.length) return;

  for (const { token } of tokens) {
    await axios.post('https://fcm.googleapis.com/fcm/send', {
      to: token,
      notification: { title, body },
      data,
    }, {
      headers: {
        'Authorization': `key=${process.env.FCM_SERVER_KEY}`,
        'Content-Type': 'application/json'
      }
    }).catch(err => console.log('FCM error:', err.message));
  }
}

module.exports = { sendPushToUser };

Usage in order controller after status update:

const { sendPushToUser } = require('../services/notification.service');

// After order delivered:
await sendPushToUser(userId, '🎉 Order Delivered!',
  `Your vegetables have been delivered. Order #${orderNumber}`,
  { orderId: String(orderId), type: 'delivery' });

Step 3 — Backend: Create Order + Verify Payment

// backend/src/routes/payment.routes.js
const express  = require('express');
const Razorpay = require('razorpay');
const crypto   = require('crypto');
const { pool } = require('../config/database');
const { protect } = require('../middleware/auth.middleware');
const router   = express.Router();

const razorpay = new Razorpay({
  key_id:     process.env.RAZORPAY_KEY_ID,
  key_secret: process.env.RAZORPAY_KEY_SECRET,
});

// POST /api/v1/payments/create-order
router.post('/create-order', protect, async (req, res) => {
  try {
    const { orderId } = req.body;
    const [rows] = await pool.execute('SELECT * FROM orders WHERE id=? AND user_id=?', [orderId, req.user.id]);
    if (!rows.length) return res.status(404).json({ success:false, message:'Order not found' });

    const order = rows[0];
    const razorOrder = await razorpay.orders.create({
      amount:   Math.round(order.total_amount * 100), // in paise
      currency: 'INR',
      receipt:  order.order_number,
      notes:    { order_number: order.order_number, user_mobile: req.user.mobile },
    });

    res.json({ success:true, data:{
      razorpayOrderId: razorOrder.id,
      amount:          razorOrder.amount,
      currency:        razorOrder.currency,
      keyId:           process.env.RAZORPAY_KEY_ID,
      orderNumber:     order.order_number,
      userName:        req.user.name,
      userMobile:      req.user.mobile,
    }});
  } catch (err) { res.status(500).json({ success:false, message:err.message }); }
});

// POST /api/v1/payments/verify
router.post('/verify', protect, async (req, res) => {
  try {
    const { razorpay_order_id, razorpay_payment_id, razorpay_signature, orderId } = req.body;

    // Verify signature (CRITICAL security step)
    const body     = razorpay_order_id + '|' + razorpay_payment_id;
    const expected = crypto.createHmac('sha256', process.env.RAZORPAY_KEY_SECRET)
                      .update(body).digest('hex');

    if (expected !== razorpay_signature) {
      return res.status(400).json({ success:false, message:'Payment verification failed' });
    }

    // Update order as paid
    await pool.execute(
      "UPDATE orders SET payment_status='paid', upi_transaction_id=?, status='confirmed' WHERE id=?",
      [razorpay_payment_id, orderId]
    );

    res.json({ success:true, message:'Payment verified! Order confirmed.' });
  } catch (err) { res.status(500).json({ success:false, message:err.message }); }
});

module.exports = router;

Razorpay has a "Payouts" product that lets you send money to any UPI ID programmatically. Cost: ₹2–5 per payout transaction. This is how you pay farmers digitally.

Option A — Manual UPI payment (Start with this, no setup needed)

When admin marks a farmer listing as "Collected":

  1. Admin sees farmer's UPI ID on the dashboard
  2. Admin opens PhonePe / GPay on their phone
  3. Sends payment to farmer's UPI ID
  4. Enters UTR (transaction reference) in admin dashboard → clicks "Mark Paid"
  5. System records payment and updates farmer's listing status to "Paid"
✅ This works today with ZERO setup. Start with this.

Option B — Automated UPI payout via Razorpay (Phase 2)

# Install Razorpay on backend (already done if you did Option A for customers)
npm install razorpay

# Add to .env:
RAZORPAY_PAYOUT_KEY=rzp_live_xxxxxx
RAZORPAY_PAYOUT_SECRET=xxxxxxxxxxxx
RAZORPAY_ACCOUNT_NUMBER=your_razorpay_current_account
// backend: Send payout to farmer UPI
const Razorpay = require('razorpay');
const client = new Razorpay({
  key_id: process.env.RAZORPAY_PAYOUT_KEY,
  key_secret: process.env.RAZORPAY_PAYOUT_SECRET,
});

async function payFarmerUPI(farmerName, upiId, amountRs, listingNumber) {
  const payout = await client.payouts.create({
    account_number: process.env.RAZORPAY_ACCOUNT_NUMBER,
    amount:         Math.round(amountRs * 100), // in paise
    currency:       'INR',
    mode:           'UPI',
    purpose:        'vendor_advance',
    fund_account: {
      account_type: 'vpa',
      vpa:          { address: upiId }, // farmer's UPI ID
      contact: {
        name:    farmerName,
        type:    'vendor',
        contact: null,
      }
    },
    narration: `Vegz payment for ${listingNumber}`,
  });
  return payout;
}
⚠️ Razorpay Payouts requires: Razorpay Current Account (open one free at razorpay.com/current-account) + add money to it + complete additional KYC. Start with manual UPI payments first.

Admin Dashboard — Credentials + Security

Create Admin Password (MUST do before going live)

The database seeds a placeholder admin. You need to set the real password hash:

# On any machine with Node.js installed:
node -e "require('bcryptjs').hash('YourNewStrongPassword@2025', 12, (err, hash) => console.log(hash))"

# Copy the hash output, then update in MySQL:
mysql -u vegz_user -p vegz_db

UPDATE admin_users 
SET password_hash='PASTE_HASH_HERE', email='admin@vegz.online', mobile='+91YOURMOBILE' 
WHERE email='admin@vegz.online';
EXIT;

Admin credentials to remember

URL:      https://admin.vegz.online
Email:    admin@vegz.online
Password: (whatever you set above)

NEVER share these with anyone.
Token expires in 12 hours — login again each day.

Security: Why other users can't access admin

  • Admin routes require a JWT token with type: "admin" — customer tokens don't work
  • Admin login is email + password — not mobile OTP like customers
  • Admin page is on a separate domain (admin.vegz.online) served from different Nginx block
  • Even if someone finds the URL, they need the email + password to get in

Step 2 — Nginx config (replace your existing /etc/nginx/sites-available/vegz)

sudo nano /etc/nginx/sites-available/vegz
# ── vegz.online — Main customer website ──────────────────
server {
    listen 80;
    server_name vegz.online www.vegz.online;
    root /var/www/vegz/frontend;
    index index.html;

    gzip on;
    gzip_types text/plain text/css application/javascript application/json;

    # Service worker must be accessible at root
    location /firebase-messaging-sw.js {
        add_header Cache-Control "no-cache";
    }

    location ~* \.(css|js|jpg|jpeg|png|gif|ico|svg|woff2)$ {
        expires 30d;
        add_header Cache-Control "public, immutable";
    }

    location / {
        try_files $uri $uri/ $uri.html /index.html;
    }
}

# ── farmer.vegz.online — Farmer portal ───────────────────
server {
    listen 80;
    server_name farmer.vegz.online;
    root /var/www/vegz/frontend;
    index pages/farmer.html;

    location / {
        try_files $uri /pages/farmer.html;
    }
}

# ── admin.vegz.online — Admin dashboard ──────────────────
server {
    listen 80;
    server_name admin.vegz.online;
    root /var/www/vegz/frontend;
    index pages/admin.html;

    # Extra security header for admin
    add_header X-Robots-Tag "noindex, nofollow" always;
    add_header X-Frame-Options "DENY" always;

    location / {
        try_files $uri /pages/admin.html;
    }
}

# ── api.vegz.online — Backend API ────────────────────────
server {
    listen 80;
    server_name api.vegz.online;
    client_max_body_size 10M;

    location / {
        proxy_pass         http://127.0.0.1:5001;
        proxy_http_version 1.1;
        proxy_set_header   Upgrade $http_upgrade;
        proxy_set_header   Connection 'upgrade';
        proxy_set_header   Host $host;
        proxy_set_header   X-Real-IP $remote_addr;
        proxy_set_header   X-Forwarded-Proto $scheme;
        proxy_read_timeout 60s;
    }
}
sudo nginx -t && sudo systemctl restart nginx
